/**
 * @param {number} n
 * @return {number[][]}
 */
var generateMatrix = function (n) {
    let arr = new Array(n); 
    for (let i = 0; i < arr.length; i++) {
        arr[i] = new Array(n).fill(0); 
    }
    let x = 0, y = 0;
    let val = 1
    function func(x, y, val, n) {
        if (n === 1) {
            arr[x][y] = val
        }
        for (let i = 0; i < n - 1; i++) {
            arr[x][y] = val
            y++
            val++
        }
        for (let j = 0; j < n - 1; j++) {
            arr[x][y] = val
            x++
            val++
        }
        for (let k = 0; k < n - 1; k++) {
            arr[x][y] = val
            y--
            val++
        }
        for (let l = 0; l < n - 1; l++) {
            arr[x][y] = val
            x--
            val++
        }
        if (n - 2 > 0) {

            func(x + 1, y + 1, val, n - 2)
        }
    }
    func(x, y, val, n)
    return arr
};
